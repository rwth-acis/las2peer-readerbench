	private static String replaceUmlaut(String input) {
 
		//replace all lower Umlauts
		String output = input.replace("Ã¼", "ue")
							 .replace("Ã¶", "oe")
							 .replace("Ã¤", "ae")
							 .replace("ÃŸ", "ss");
	
		//first replace all capital umlaute in a non-capitalized context (e.g. Ãœbung)
		output = output.replaceAll("Ãœ(?=[a-zÃ¤Ã¶Ã¼ÃŸ ])", "Ue")
					   .replaceAll("Ã–(?=[a-zÃ¤Ã¶Ã¼ÃŸ ])", "Oe")
					   .replaceAll("Ã„(?=[a-zÃ¤Ã¶Ã¼ÃŸ ])", "Ae");
	
		//now replace all the other capital umlaute
		output = output.replace("Ãœ", "UE")
					   .replace("Ã–", "OE")
					   .replace("Ã„", "AE");
	
		return output;
	}
